  success: boolean
  data?: T
  message?: string
  errors?: string[]
}

// Helper function
function sendSuccess<T>(res: Response, data: T, code = 200) {
  return res.status(code).json({
    success: true, data
  } satisfies ApiResponse<T>)
}

function sendError(res: Response, msg: string, code = 400) {
  return res.status(code).json({
    success: false, message: msg
  })
}

🔧 สร้าง Middleware เอง

import { Request, Response, NextFunction } from 'express'

// ─── Logger Middleware ─────────────────────────────
export function requestLogger(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const start = Date.now()
  console.log(`→ ${req.method} ${req.path}`)

  res.on('finish', () => {
    const ms = Date.now() - start
    console.log(`← ${res.statusCode} (${ms}ms)`)
  })

  next()  // ส่งต่อไป middleware ถัดไป
}

// ─── Rate Limiter Middleware ───────────────────────
const requestCounts = new Map<string, number[]>()

export function rateLimiter(
  max: number,
  windowMs: number
) {
  return (req: Request, res: Response, next: NextFunction) => {
    const ip = req.ip ?? 'unknown'
    const now = Date.now()
    const times = (requestCounts.get(ip) ?? [])
      .filter(t => now - t < windowMs)

    if (times.length >= max) {
      return res.status(429).json({ message: 'Too Many Requests' })
    }

    times.push(now)
    requestCounts.set(ip, times)
    next()
  }
}

🔒 Auth Middleware (JWT)

import { Request, Response, NextFunction } from 'express'
import jwt from 'jsonwebtoken'

interface JwtPayload {
  id: number
  email: string
  role: string
}

export function authenticate(
  req: Request,
  res: Response,
  next: NextFunction
) {
  try {
    const header = req.headers.authorization
    if (!header?.startsWith('Bearer ')) {
      return res.status(401).json({ message: 'ไม่มี Token' })
    }

    const token = header.slice(7)
    const decoded = jwt.verify(
      token, process.env.JWT_SECRET!
    ) as JwtPayload

    req.user = decoded  // ผ่าน extended type
    next()
  } catch {
    res.status(401).json({ message: 'Token ไม่ถูกต้อง' })
  }
}

// Role-based middleware factory
export function requireRole(...roles: string[]) {
  return (req: Request, res: Response, next: NextFunction) => {
    if (!roles.includes(req.user?.role ?? '')) {
      return res.status(403).json({ message: 'ไม่มีสิทธิ์' })
    }
    next()
  }
}

// ใช้งาน
router.delete('/:id', authenticate, requireRole('admin'), handler)

⚙️ User Service

// src/services/user.service.ts
import bcrypt from 'bcrypt'
import { userRepository } from '@/repositories/user.repository'
import { CreateUserDto } from '@/types/user.types'
import { AppError } from '@/utils/AppError'

interface FindAllOptions {
  page: number; limit: number; search?: string
}

export const userService = {
  async findAll({ page, limit, search }: FindAllOptions) {
    const skip = (page - 1) * limit
    const [users, total] = await userRepository.findAndCount({ skip, limit, search })
    return { data: users, total, page, totalPages: Math.ceil(total / limit) }
  },

  async findById(id: number) {
    return userRepository.findById(id)
  },

  async create(dto: CreateUserDto) {
    // ตรวจ email ซ้ำ
    const existing = await userRepository.findByEmail(dto.email)
    if (existing) throw new AppError('Email นี้ถูกใช้แล้ว', 409)

    // hash password
    const hashed = await bcrypt.hash(dto.password, 12)
    return userRepository.create({ ...dto, password: hashed })
  },

  async delete(id: number) {
    const user = await userRepository.findById(id)
    if (!user) throw new AppError('ไม่พบผู้ใช้', 404)
    return userRepository.delete(id)
  }
}

🔧 Validate Middleware

// src/middleware/validate.ts
import { Request, Response, NextFunction } from 'express'
import { AnyZodObject, ZodError } from 'zod'

export function validate(schema: AnyZodObject) {
  return async (req: Request, res: Response, next: NextFunction) => {
    try {
      // parse และ transform ข้อมูลใน place
      const parsed = await schema.parseAsync({
        body:   req.body,
        query:  req.query,
        params: req.params,
      })

      // แทนที่ด้วยข้อมูลที่ผ่านการ transform แล้ว
      req.body   = parsed.body   ?? req.body
      req.query  = parsed.query  ?? req.query
      req.params = parsed.params ?? req.params

      next()
    } catch (err) {
      if (err instanceof ZodError) {
        return res.status(422).json({
          success: false,
          message: 'ข้อมูลไม่ถูกต้อง',
          errors: err.errors.map(e => ({
            field: e.path.join('.'),
            message: e.message,
          }))
        })
      }
      next(err)
    }
  }
}

// src/services/auth.service.ts
import bcrypt from 'bcrypt'
import jwt from 'jsonwebtoken'
import { userRepository } from '@/repositories/user.repository'
import { AppError } from '@/utils/AppError'

const ACCESS_SECRET  = process.env.JWT_SECRET!
const REFRESH_SECRET = process.env.JWT_REFRESH_SECRET!

export interface TokenPayload {
  id: number; email: string; role: string
}

function signTokens(payload: TokenPayload) {
  const accessToken = jwt.sign(payload, ACCESS_SECRET, {
    expiresIn: '15m'
  })
  const refreshToken = jwt.sign(payload, REFRESH_SECRET, {
    expiresIn: '7d'
  })
  return { accessToken, refreshToken }
}

export const authService = {
  async login(email: string, password: string) {
    const user = await userRepository.findByEmail(email)
    if (!user) throw new AppError('Email หรือรหัสผ่านไม่ถูกต้อง', 401)

    const valid = await bcrypt.compare(password, user.password)
    if (!valid) throw new AppError('Email หรือรหัสผ่านไม่ถูกต้อง', 401)

    const tokens = signTokens({ id: user.id, email: user.email, role: user.role })

    // เก็บ refreshToken ใน DB (optional)
    await userRepository.updateRefreshToken(user.id, tokens.refreshToken)

    const { password: _, ...userWithoutPw } = user
    return { user: userWithoutPw, ...tokens }
  },

  async refreshToken(token: string) {
    try {
      const payload = jwt.verify(token, REFRESH_SECRET) as TokenPayload
      return signTokens({ id: payload.id, email: payload.email, role: payload.role })
    } catch {
      throw new AppError('Refresh token ไม่ถูกต้อง', 401)
    }
  },

  async logout(userId: number) {
    await userRepository.updateRefreshToken(userId, null)
  }
}

🎮 Auth Controller

// src/controllers/auth.controller.ts
import { Request, Response, NextFunction } from 'express'
import { authService } from '@/services/auth.service'

export async function login(req: Request, res: Response, next: NextFunction) {
  try {
    const { email, password } = req.body
    const result = await authService.login(email, password)

    // เก็บ refreshToken ใน httpOnly cookie (ปลอดภัยกว่า localStorage)
    res.cookie('refreshToken', result.refreshToken, {
      httpOnly: true,
      secure: process.env.NODE_ENV === 'production',
      sameSite: 'strict',
      maxAge: 7 * 24 * 60 * 60 * 1000  // 7 วัน
    })

    res.json({
      success: true,
      user: result.user,
      accessToken: result.accessToken
    })
  } catch (err) { next(err) }
}

export async function refresh(req: Request, res: Response, next: NextFunction) {
  try {
    const token = req.cookies?.refreshToken
    if (!token) return res.status(401).json({ message: 'ไม่มี refresh token' })

    const tokens = await authService.refreshToken(token)
    res.json({ success: true, ...tokens })
  } catch (err) { next(err) }
}

export async function logout(req: Request, res: Response, next: NextFunction) {
  try {
    await authService.logout(req.user!.id)
    res.clearCookie('refreshToken')
    res.json({ success: true, message: 'ออกจากระบบสำเร็จ' })
  } catch (err) { next(err) }
}

🏪 Repository Pattern

// src/repositories/user.repository.ts
import { prisma } from '@/lib/prisma'
import type { Prisma } from '@prisma/client'

export const userRepository = {
  async findAndCount({ skip, limit, search }: {
    skip: number; limit: number; search?: string
  }) {
    const where: Prisma.UserWhereInput = search
      ? { OR: [
          { name:  { contains: search, mode: 'insensitive' } },
          { email: { contains: search, mode: 'insensitive' } },
        ]}
      : {}

    return prisma.$transaction([
      prisma.user.findMany({
        where, skip, take: limit,
        select: { id: true, name: true, email: true, role: true, createdAt: true },
        orderBy: { createdAt: 'desc' }
      }),
      prisma.user.count({ where })
    ])
  },

  async findById(id: number) {
    return prisma.user.findUnique({
      where: { id },
      select: { id: true, name: true, email: true, role: true, profile: true }
    })
  },

  async findByEmail(email: string) {
    return prisma.user.findUnique({ where: { email } })
  },

  async create(data: Prisma.UserCreateInput) {
    return prisma.user.create({
      data,
      select: { id: true, name: true, email: true, role: true }
    })
  },

  async update(id: number, data: Prisma.UserUpdateInput) {
    return prisma.user.update({ where: { id }, data })
  },

  async delete(id: number) {
    return prisma.user.delete({ where: { id } })
  },

  async updateRefreshToken(id: number, token: string | null) {
    return prisma.user.update({ where: { id }, data: { refreshToken: token } })
  }
}

🔴 Custom AppError Class

// src/utils/AppError.ts
export class AppError extends Error {
  public readonly statusCode: number
  public readonly isOperational: boolean

  constructor(message: string, statusCode = 500, isOperational = true) {
    super(message)
    this.statusCode = statusCode
    this.isOperational = isOperational
    Error.captureStackTrace(this, this.constructor)
    Object.setPrototypeOf(this, AppError.prototype)
  }
}

// ─── Specific Error Classes ────────────────────────
export class NotFoundError extends AppError {
  constructor(resource = 'ข้อมูล') {
    super(`ไม่พบ ${resource}`, 404)
  }
}

export class UnauthorizedError extends AppError {
  constructor(message = 'กรุณาเข้าสู่ระบบ') {
    super(message, 401)
  }
}

export class ForbiddenError extends AppError {
  constructor(message = 'ไม่มีสิทธิ์เข้าถึง') {
    super(message, 403)
  }
}

export class ConflictError extends AppError {
  constructor(message: string) {
    super(message, 409)
  }
}

// ใช้งาน
throw new NotFoundError('ผู้ใช้')     // 404: ไม่พบผู้ใช้
throw new ConflictError('Email ซ้ำ') // 409: Email ซ้ำ

🛡️ Global Error Handler Middleware

// src/middleware/errorHandler.ts
import { Request, Response, NextFunction } from 'express'
import { ZodError } from 'zod'
import { Prisma } from '@prisma/client'
import { AppError } from '@/utils/AppError'

export function errorHandler(
  err: unknown,
  req: Request,
  res: Response,
  _next: NextFunction
) {
  // 1. AppError (ที่เราสร้างเอง)
  if (err instanceof AppError) {
    return res.status(err.statusCode).json({
      success: false,
      message: err.message
    })
  }

  // 2. Zod Validation Error
  if (err instanceof ZodError) {
    return res.status(422).json({
      success: false,
      message: 'ข้อมูลไม่ถูกต้อง',
      errors: err.errors.map(e => ({
        field: e.path.join('.'),
        message: e.message
      }))
    })
  }

  // 3. Prisma Errors
  if (err instanceof Prisma.PrismaClientKnownRequestError) {
    if (err.code === 'P2002') {  // Unique constraint
      const field = (err.meta?.target as string[])?.join(', ')
      return res.status(409).json({
        success: false, message: `${field} นี้ถูกใช้ไปแล้ว`
      })
    }
    if (err.code === 'P2025') {  // Record not found
      return res.status(404).json({
        success: false, message: 'ไม่พบข้อมูลที่ต้องการ'
      })
    }
  }

  // 4. Unknown error — ไม่เปิดเผย detail ใน production
  console.error('Unhandled error:', err)
  res.status(500).json({
    success: false,
    message: process.env.NODE_ENV === 'production'
      ? 'เกิดข้อผิดพลาดภายในเซิร์ฟเวอร์'
      : (err instanceof Error ? err.message : 'Unknown error')
  })
}

🔄 Async Handler Wrapper

// src/utils/asyncHandler.ts
// ห่อ async functions เพื่อ catch errors อัตโนมัติ
import { Request, Response, NextFunction, RequestHandler } from 'express'

type AsyncFn = (req: Request, res: Response, next: NextFunction) => Promise<any>

export function asyncHandler(fn: AsyncFn): RequestHandler {
  return (req, res, next) => {
    Promise.resolve(fn(req, res, next)).catch(next)
  }
}

// ─── ก่อนใช้ asyncHandler ────────────────────────
router.get('/', async (req, res, next) => {
  try {
    const data = await someAsyncOp()
    res.json(data)
  } catch (err) { next(err) }  // ต้องเขียน try/catch ทุกครั้ง
})

// ─── หลังใช้ asyncHandler ────────────────────────
router.get('/', asyncHandler(async (req, res) => {
  const data = await someAsyncOp()
  res.json(data)
  // error จะถูก catch อัตโนมัติ ✅
}))

🔒 Validate Environment Variables

// src/config/env.ts
import { z } from 'zod'
import 'dotenv/config'

const envSchema = z.object({
  NODE_ENV:           z.enum(['development', 'production', 'test'])
                       .default('development'),
  PORT:               z.string().default('3000'),
  DATABASE_URL:       z.string().url(),
  JWT_SECRET:         z.string().min(32),
  JWT_REFRESH_SECRET: z.string().min(32),
  CORS_ORIGIN:        z.string().default('*'),
})

const parsed = envSchema.safeParse(process.env)

if (!parsed.success) {
  console.error('❌ Invalid environment variables:')
  console.error(parsed.error.flatten().fieldErrors)
  process.exit(1)  // หยุดทำงานทันที
}

export const env = parsed.data

// ใช้งาน
import { env } from '@/config/env'
console.log(env.PORT)  // type-safe ✅

✅ Best Practices

  • ใช้ Controller → Service → Repository เสมอ
  • Throw AppError แทน res.status() ใน service
  • ใช้ asyncHandler ครอบทุก async route
  • Validate env vars ตอน startup
  • เก็บ refreshToken ใน httpOnly cookie
  • ใช้ select ใน Prisma เสมอ — ไม่ return password
  • ใช้ helmet() และ cors() ทุกโปรเจกต์
  • ใช้ Zod infer สร้าง DTO types

❌ หลีกเลี่ยง

  • ไม่เขียน logic ใน routes โดยตรง
  • ไม่ใช้ any type
  • ไม่ return password ใน response
  • ไม่ตรวจสอบ input ใน controller (ให้ validate middleware ทำ)
  • ไม่ใส่ secret ใน code — ใช้ .env
  • ไม่ commit .env ขึ้น git
  • ไม่ใช้ console.log ใน production
  • ไม่ทำ database query ใน controller
