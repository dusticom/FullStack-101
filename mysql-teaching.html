บทที่ 5 — CRUD และ MySQL-specific Syntax

AUTO_INCREMENT และ LAST_INSERT_ID()

CREATE TABLE users ( id INT UNSIGNED NOT NULL AUTO_INCREMENT, name VARCHAR(100), PRIMARY KEY (id) ) ENGINE=InnoDB AUTO_INCREMENT=1000; -- เริ่มจาก 1000 INSERT INTO users (name) VALUES ('สมชาย'); -- ดู id ที่เพิ่งใส่ไป SELECT LAST_INSERT_ID(); -- 1000 -- รีเซ็ต AUTO_INCREMENT ALTER TABLE users AUTO_INCREMENT = 1;

ON DUPLICATE KEY UPDATE (Upsert)

-- ถ้า primary/unique key ซ้ำ → UPDATE แทน INSERT INSERT INTO page_views (page_id, views) VALUES (42, 1) ON DUPLICATE KEY UPDATE views = views + 1, last_viewed = NOW(); -- ใช้ VALUES() อ้างอิงค่าที่จะ INSERT INSERT INTO product_stock (product_id, stock) VALUES (1, 100) ON DUPLICATE KEY UPDATE stock = VALUES(stock); -- ใช้ค่า 100 ที่ระบุ

Multi-table UPDATE และ DELETE

-- UPDATE หลายตารางพร้อมกัน UPDATE orders o JOIN customers c ON o.customer_id = c.id SET o.discount = 0.1, c.tier = 'gold' WHERE c.total_spent > 100000; -- DELETE จากหลายตารางพร้อมกัน DELETE o, oi FROM orders o JOIN order_items oi ON o.id = oi.order_id WHERE o.status = 'cancelled' AND o.order_date < '2023-01-01';

EXPLAIN — วิเคราะห์ Query Plan

EXPLAIN SELECT * FROM orders o JOIN customers c ON o.customer_id = c.id WHERE o.status = 'paid'; /* id | select_type | table | type | key | rows | Extra 1 | SIMPLE | o | ref | idx_status | 500 | Using index 1 | SIMPLE | c | eq_ref| PRIMARY | 1 | type ยิ่งดีขึ้น: ALL(Full Scan) → index → range → ref → eq_ref → const/system */ -- EXPLAIN ANALYZE (MySQL 8.0.18+) EXPLAIN ANALYZE SELECT * FROM orders WHERE customer_id = 5; -- แสดงเวลาจริงที่ใช้แต่ละ step

Covering Index และ Composite Index

-- Composite Index: ลำดับ column สำคัญมาก! -- Rule: (Equality → Range → Sort → Select) CREATE INDEX idx_order_lookup ON orders (customer_id, status, order_date); -- index นี้ช่วย query นี้ได้เต็มที่: WHERE customer_id = 5 AND status = 'paid' ORDER BY order_date; -- Covering Index: query ดึงแค่ cols ใน index CREATE INDEX idx_cover ON orders (customer_id, total, status); -- ไม่ต้อง access ตาราง (Extra: Using index) SELECT total, status FROM orders WHERE customer_id = 5;

Full-text Search

CREATE TABLE articles ( id INT PRIMARY KEY AUTO_INCREMENT, title VARCHAR(255), content TEXT, FULLTEXT INDEX ft_idx (title, content) ); -- Natural Language Mode (default) SELECT *, MATCH(title, content) AGAINST ('MySQL database') AS score FROM articles WHERE MATCH(title, content) AGAINST ('MySQL database') ORDER BY score DESC; -- Boolean Mode (+ ต้องมี, - ต้องไม่มี, * wildcard) WHERE MATCH(title, content) AGAINST ('+MySQL +performance -slow' IN BOOLEAN MODE);

บทที่ 9 — Transactions, Locks และ Deadlocks

COMMIT/ROLLBACK, Row Locks, Deadlock detection

Locking ใน InnoDB

Lock Typeใครถือได้เกิดเมื่อ
Shared Lock (S)หลาย transactionSELECT ... FOR SHARE
Exclusive Lock (X)1 transaction เท่านั้นUPDATE, DELETE, SELECT ... FOR UPDATE
Intention Shared (IS)Table levelก่อน S lock บน row
Intention Exclusive (IX)Table levelก่อน X lock บน row

SELECT FOR UPDATE และ FOR SHARE

-- Lock แถวเพื่อ update (ป้องกัน race condition) START TRANSACTION; SELECT stock FROM products WHERE id = 101 FOR UPDATE; -- ← lock แถวนี้ไว้ -- ตรวจ stock และ update UPDATE products SET stock = stock - 1 WHERE id = 101; COMMIT; -- SKIP LOCKED: ข้ามแถวที่ถูก lock ไปแล้ว SELECT * FROM job_queue WHERE status = 'pending' LIMIT 10 FOR UPDATE SKIP LOCKED; -- ← สำหรับ Queue processing

Deadlock และการป้องกัน

Deadlock เกิดขึ้น
-- TX A: UPDATE accounts SET... WHERE id=1; -- (รอ TX B ปล่อย lock id=2) UPDATE accounts SET... WHERE id=2; -- TX B (พร้อมกัน): UPDATE accounts SET... WHERE id=2; -- (รอ TX A ปล่อย lock id=1) UPDATE accounts SET... WHERE id=1;
แก้: Lock ตามลำดับเดียวกัน
-- ทั้ง TX A และ TX B: -- lock id เล็กก่อนเสมอ UPDATE accounts SET... WHERE id = LEAST(1,2); UPDATE accounts SET... WHERE id = GREATEST(1,2);
-- ดู deadlock ล่าสุด SHOW ENGINE INNODB STATUS\G -- ตั้งค่า timeout (default 50 วินาที) SET innodb_lock_wait_timeout = 10;

บทที่ 10 — Users, Permissions และ Security

จัดการ Users

-- สร้าง User CREATE USER 'appuser'@'localhost' IDENTIFIED BY 'StrongP@ssw0rd!'; -- User จาก host ใดก็ได้ CREATE USER 'reader'@'%' IDENTIFIED BY 'password'; -- เปลี่ยน password ALTER USER 'appuser'@'localhost' IDENTIFIED BY 'NewPassword!'; -- ลบ User DROP USER 'olduser'@'localhost'; -- ดู Users ทั้งหมด SELECT user, host, authentication_string FROM mysql.user;

GRANT และ REVOKE

-- ให้ทุก permission บน database เดียว GRANT ALL PRIVILEGES ON myapp.* TO 'appuser'@'localhost'; -- ให้แค่ SELECT GRANT SELECT ON myapp.* TO 'reader'@'%'; -- ให้ SELECT, INSERT, UPDATE บนตารางเดียว GRANT SELECT, INSERT, UPDATE ON myapp.orders TO 'appuser'@'localhost'; -- ถอน permission REVOKE INSERT ON myapp.* FROM 'reader'@'%'; -- apply ทันที FLUSH PRIVILEGES; -- ดู permissions ของ user SHOW GRANTS FOR 'appuser'@'localhost';
Principle of Least Privilege
App user ควรมีแค่ SELECT, INSERT, UPDATE, DELETE บน database ที่ใช้งาน — ไม่ควรใช้ root สำหรับ application

Replication (Master-Slave)

1
Master — เขียน changes ไปยัง Binary Log (binlog)
→ Binlog Stream →
2
Slave I/O Thread — อ่าน binlog จาก master, เขียนไป Relay Log
↓
3
Slave SQL Thread — อ่าน Relay Log และ apply กับ Slave database
ประโยชน์ของ Replication
  • Read Scaling — ส่ง SELECT queries ไป Slave ลด load บน Master
  • Backup — ทำ backup บน Slave โดยไม่กระทบ Master
  • High Availability — ถ้า Master ล่ม สามารถ promote Slave เป็น Master

Checklist — MySQL Best Practices

✅ ควรทำ
  • ✅ ใช้ InnoDB engine เสมอ
  • ✅ ใช้ utf8mb4 charset
  • ✅ ใส่ index บน FK, WHERE, JOIN columns
  • ✅ ใช้ DECIMAL สำหรับเงิน
  • ✅ ใช้ prepared statements ป้องกัน SQL Injection
  • ✅ ตั้ง innodb_buffer_pool_size ให้เหมาะสม
  • ✅ เปิด slow query log
  • ✅ Backup สม่ำเสมอและทดสอบ restore
  • ✅ ใช้ EXPLAIN วิเคราะห์ query ช้า
  • ✅ Principle of least privilege สำหรับ users
❌ ควรหลีกเลี่ยง
  • ❌ ใช้ root user สำหรับ application
  • ❌ SELECT * ใน production
  • ❌ FLOAT/DOUBLE สำหรับเงิน
  • ❌ ไม่มี WHERE ใน UPDATE/DELETE
  • ❌ Index ทุก column (ช้า write)
  • ❌ N+1 query problem
  • ❌ ฝัง password ใน code
  • ❌ ไม่ backup
  • ❌ LIKE '%xxx' บน indexed column
  • ❌ transaction ขนาดใหญ่เกินไป
