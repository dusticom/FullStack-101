  items: T[]
  renderItem: (item: T, index: number) => React.ReactNode
  keyExtractor: (item: T) => string
}

  )
}

// src/store/features/counterSlice.ts
import { createSlice, PayloadAction } from '@reduxjs/toolkit'

interface CounterState {
  value: number
  status: 'idle' | 'loading' | 'failed'
}

const initialState: CounterState = {
  value: 0,
  status: 'idle',
}

export const counterSlice = createSlice({
  name: 'counter',
  initialState,
  reducers: {
    // Immer ทำให้เขียน "mutate" ได้ แต่จริงๆ immutable
    increment: (state) => {
      state.value += 1
    },
    decrement: (state) => {
      state.value -= 1
    },
    incrementByAmount: (state, action: PayloadAction<number>) => {
      state.value += action.payload
    },
    reset: (state) => {
      state.value = 0
    },
  },
})

// Export actions
export const { increment, decrement, incrementByAmount, reset } =
  counterSlice.actions

// Export reducer
export default counterSlice.reducer

// Selectors
export const selectCount = (state: RootState) => state.counter.value

⚡ createAsyncThunk — Async Actions

// src/store/features/userSlice.ts
import { createSlice, createAsyncThunk } from '@reduxjs/toolkit'

interface User { id: number; name: string; email: string }
interface UserState {
  users: User[]
  loading: boolean
  error: string | null
}

// Async thunk — fetch users จาก API
export const fetchUsers = createAsyncThunk(
  'users/fetchAll',
  async (_, { rejectWithValue }) => {
    try {
      const res = await fetch('/api/users')
      if (!res.ok) throw new Error('Network error')
      return (await res.json()) as User[]
    } catch (err) {
      return rejectWithValue((err as Error).message)
    }
  }
)

const userSlice = createSlice({
  name: 'users',
  initialState: { users: [], loading: false, error: null } as UserState,
  reducers: {},
  extraReducers: (builder) => {
    builder
      .addCase(fetchUsers.pending, (state) => {
        state.loading = true; state.error = null
      })
      .addCase(fetchUsers.fulfilled, (state, { payload }) => {
        state.loading = false; state.users = payload
      })
      .addCase(fetchUsers.rejected, (state, { payload }) => {
        state.loading = false
        state.error = payload as string
      })
  },
})

export default userSlice.reducer

🛒 ตัวอย่างจริง: Cart Slice

// src/store/features/cartSlice.ts
import { createSlice, PayloadAction } from '@reduxjs/toolkit'

interface CartItem {
  id: number; name: string; price: number; qty: number
}
interface CartState { items: CartItem[] }

const cartSlice = createSlice({
  name: 'cart',
  initialState: { items: [] } as CartState,
  reducers: {
    addItem: (state, { payload }: PayloadAction<CartItem>) => {
      const existing = state.items.find((i) => i.id === payload.id)
      if (existing) existing.qty += 1
      else state.items.push({ ...payload, qty: 1 })
    },
    removeItem: (state, { payload }: PayloadAction<number>) => {
      state.items = state.items.filter((i) => i.id !== payload)
    },
    updateQty: (state, { payload }: PayloadAction<{ id: number; qty: number }>) => {
      const item = state.items.find((i) => i.id === payload.id)
      if (item) item.qty = payload.qty
    },
    clearCart: (state) => { state.items = [] },
  },
})

export const { addItem, removeItem, updateQty, clearCart } = cartSlice.actions
export const selectCartTotal = (state: RootState) =>
  state.cart.items.reduce((sum, i) => sum + i.price * i.qty, 0)
export default cartSlice.reducer

🚀 RTK Query — Data Fetching + Caching

// src/store/services/apiSlice.ts
import { createApi, fetchBaseQuery } from '@reduxjs/toolkit/query/react'

interface Post { id: number; title: string; body: string }

export const apiSlice = createApi({
  reducerPath: 'api',
  baseQuery: fetchBaseQuery({ baseUrl: '/api' }),
  tagTypes: ['Post', 'User'],
  endpoints: (builder) => ({
    // GET /api/posts
    getPosts: builder.query<Post[], void>({
      query: () => '/posts',
      providesTags: ['Post'],
    }),
    // GET /api/posts/:id
    getPost: builder.query<Post, number>({
      query: (id) => `/posts/${id}`,
    }),
    // POST /api/posts
    createPost: builder.mutation<Post, Omit<Post, 'id'>>({
      query: (body) => ({ url: '/posts', method: 'POST', body }),
      invalidatesTags: ['Post'],  // auto refetch
    }),
    // DELETE /api/posts/:id
    deletePost: builder.mutation<void, number>({
      query: (id) => ({ url: `/posts/${id}`, method: 'DELETE' }),
      invalidatesTags: ['Post'],
    }),
  }),
})

export const {
  useGetPostsQuery,
  useGetPostQuery,
  useCreatePostMutation,
  useDeletePostMutation,
} = apiSlice

// เพิ่ม api reducer ใน store
// reducer: { ..., [apiSlice.reducerPath]: apiSlice.reducer }
// middleware: (getDefaultMiddleware) => getDefaultMiddleware().concat(apiSlice.middleware)

💡 ใช้ RTK Query ใน Component

"use client"
import { useGetPostsQuery, useDeletePostMutation } from '@/store/services/apiSlice'

export default function PostList() {
  // Auto fetch, cache, loading/error states
  const { data: posts, isLoading, isError, refetch } = useGetPostsQuery()
  const [deletePost, { isLoading: isDeleting }] = useDeletePostMutation()

  if (isLoading) return <p>กำลังโหลด...</p>
  if (isError)   return <p>เกิดข้อผิดพลาด <button onClick={refetch}>ลองใหม่</button></p>

  return (
    <ul>
      {posts?.map((post) => (
        <li key={post.id}>
          {post.title}
          <button
            onClick={() => deletePost(post.id)}
            disabled={isDeleting}
          >
            ลบ
          </button>
        </li>
      ))}
    </ul>
  )
}

✅ ทำ

  • ใช้ useAppDispatch / useAppSelector แทน hooks ทั่วไป
  • แยก selector ออกมาเป็น function ต่างหาก
  • ใช้ RTK Query แทนการ fetch เองเสมอเมื่อทำได้
  • กำหนด TypeScript types ทุกที่
  • ใช้ Server Components สำหรับ static data
  • เก็บ RootState type ไว้ที่เดียว
  • ใช้ createEntityAdapter สำหรับ list data

❌ หลีกเลี่ยง

  • ไม่เก็บ derived state ที่คำนวณได้จาก state อื่น
  • ไม่ dispatch action ใน reducer
  • ไม่ทำ async ใน reducer (ใช้ thunk แทน)
  • ไม่ mutate state โดยตรง (นอก createSlice)
  • ไม่ใส่ non-serializable values ใน store
  • ไม่ fetch ข้อมูลใน Server Component แล้วส่งผ่าน Redux
  • ไม่ใช้ any type ใน TypeScript

🔐 Custom Hook Pattern

// hooks/useCart.ts — ห่อ logic ไว้ที่เดียว
"use client"
import { useAppDispatch, useAppSelector } from '@/store'
import { addItem, removeItem, clearCart, selectCartTotal } from '@/store/features/cartSlice'
import type { CartItem } from '@/types'

export function useCart() {
  const dispatch = useAppDispatch()
  const items    = useAppSelector(s => s.cart.items)
  const total    = useAppSelector(selectCartTotal)

  return {
    items,
    total,
    itemCount: items.length,
    addItem:   (item: CartItem) => dispatch(addItem(item)),
    removeItem:(id: number)    => dispatch(removeItem(id)),
    clearCart: ()               => dispatch(clearCart()),
  }
}

// ใช้งาน — สะอาดกว่ามาก
const { items, total, addItem } = useCart()
