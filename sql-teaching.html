โครงสร้างคำสั่ง SELECT

syntax
SELECT -- 5. เลือก columns ที่ต้องการ FROM -- 1. ระบุตาราง JOIN -- 2. เชื่อมตาราง WHERE -- 3. กรองแถว GROUP BY -- 4. จัดกลุ่ม HAVING -- 5. กรองกลุ่ม ORDER BY -- 6. เรียงลำดับ LIMIT -- 7. จำกัดจำนวน
Logical Query Processing Order
ลำดับที่ SQL ประมวลผลจริง (ซ้าย→ขวา) คือ FROM → JOIN → WHERE → GROUP BY → HAVING → SELECT → ORDER BY → LIMIT ซึ่งต่างจากลำดับที่เขียน — นี่คือสาเหตุที่ alias ใน SELECT ใช้ใน ORDER BY ได้ แต่ใช้ใน WHERE ไม่ได้

SELECT ทั่วไป

-- ดูทุก column SELECT * FROM customers; -- เลือก column ที่ต้องการ SELECT name, email, city FROM customers; -- ตั้งชื่อ Alias SELECT name AS customer_name, email AS contact_email FROM customers; -- คำนวณใน SELECT SELECT name, price, price * 1.07 AS price_with_vat FROM products;

ORDER BY, LIMIT, DISTINCT

-- เรียงจากน้อยไปมาก (default) SELECT * FROM products ORDER BY price ASC; -- เรียงจากมากไปน้อย SELECT * FROM products ORDER BY price DESC; -- เรียงหลาย column SELECT * FROM products ORDER BY category ASC, price DESC; -- จำกัดจำนวนผลลัพธ์ SELECT * FROM products ORDER BY price DESC LIMIT 10; -- OFFSET (Pagination) SELECT * FROM products ORDER BY id LIMIT 10 OFFSET 20; -- หน้า 3 -- ค่าไม่ซ้ำ SELECT DISTINCT city FROM customers;

บทที่ 3 — WHERE และการกรองข้อมูล

Operators, LIKE, IN, BETWEEN, NULL handling

Comparison Operators

SELECT * FROM products WHERE price > 100; SELECT * FROM products WHERE price >= 100 AND price <= 500; SELECT * FROM orders WHERE status = 'paid'; SELECT * FROM orders WHERE status != 'cancelled'; -- หรือ <>

Logical Operators

-- AND ต้องเป็นจริงทั้งคู่ WHERE category = 'Electronics' AND price < 5000; -- OR เป็นจริงอย่างน้อยหนึ่ง WHERE city = 'Bangkok' OR city = 'Chiang Mai'; -- NOT ตรงข้าม WHERE NOT status = 'cancelled'; -- การใช้วงเล็บ WHERE (city = 'Bangkok' OR city = 'Phuket') AND total > 1000;

BETWEEN, IN, LIKE

-- BETWEEN (inclusive) WHERE price BETWEEN 100 AND 500; WHERE order_date BETWEEN '2024-01-01' AND '2024-12-31'; -- IN (เช็คหลายค่า) WHERE city IN ('Bangkok', 'Phuket', 'CM'); WHERE status NOT IN ('cancelled', 'pending'); -- LIKE (Pattern matching) WHERE name LIKE '%phone%'; -- มี phone WHERE email LIKE '%@gmail.com'; WHERE name LIKE 'A%'; -- ขึ้นต้นด้วย A WHERE code LIKE 'A__'; -- A + 2 ตัวใดก็ได้

NULL Handling

-- ตรวจสอบ NULL ต้องใช้ IS NULL WHERE email IS NULL; WHERE email IS NOT NULL; -- ❌ ผิด! = NULL ไม่ทำงาน WHERE email = NULL; -- always FALSE -- COALESCE: คืนค่าแรกที่ไม่ใช่ NULL SELECT COALESCE(phone, email, 'N/A') AS contact FROM customers; -- NULLIF: คืน NULL ถ้าสองค่าเท่ากัน SELECT NULLIF(score, 0) FROM results; -- ป้องกัน division by zero SELECT total / NULLIF(qty, 0) FROM sales;

CASE WHEN — เงื่อนไขใน SELECT

SELECT name, price, CASE WHEN price < 100 THEN 'ราคาถูก' WHEN price < 1000 THEN 'ราคากลาง' ELSE 'ราคาแพง' END AS price_range, CASE status WHEN 'paid' THEN '✅ ชำระแล้ว' WHEN 'pending' THEN '⏳ รอชำระ' ELSE '❌ ยกเลิก' END AS status_th FROM products;

INSERT, UPDATE, DELETE และ Upsert

INSERT

-- แทรกแถวเดียว INSERT INTO customers (name, email, city) VALUES ('สมชาย ใจดี', 'somchai@email.com', 'Bangkok'); -- แทรกหลายแถวพร้อมกัน INSERT INTO products (name, price, category) VALUES ('iPhone 15', 35990, 'Electronics'), ('AirPods Pro', 9990, 'Electronics'), ('MacBook Air', 44900, 'Electronics'); -- INSERT จาก SELECT INSERT INTO archive_orders SELECT * FROM orders WHERE order_date < '2023-01-01';

UPDATE และ DELETE

-- UPDATE แถวเดียว UPDATE customers SET city = 'Chiang Mai' WHERE id = 5; -- UPDATE หลาย column UPDATE products SET price = price * 0.9, -- ลด 10% stock = stock - 1 WHERE category = 'Electronics'; -- DELETE แถวที่ต้องการ DELETE FROM orders WHERE status = 'cancelled' AND order_date < '2023-01-01';

UPSERT — INSERT or UPDATE

-- MySQL: ON DUPLICATE KEY UPDATE INSERT INTO product_stats (product_id, views) VALUES (101, 1) ON DUPLICATE KEY UPDATE views = views + 1; -- PostgreSQL: ON CONFLICT DO UPDATE INSERT INTO product_stats (product_id, views) VALUES (101, 1) ON CONFLICT (product_id) DO UPDATE SET views = product_stats.views + 1;
ระวัง!
DELETE FROM table โดยไม่มี WHERE จะลบทุกแถว! ควรทดสอบด้วย SELECT ก่อน แล้วค่อยเปลี่ยนเป็น DELETE

บทที่ 6 — JOINs การเชื่อมตาราง

INNER JOIN, LEFT/RIGHT JOIN, FULL OUTER JOIN, SELF JOIN, CROSS JOIN

ประเภท JOIN

JOINผลลัพธ์
INNER JOINแถวที่มีคู่ match ในทั้งสองตาราง
LEFT JOINทุกแถวของตารางซ้าย + ข้อมูลตารางขวาที่ match (NULL ถ้าไม่มี)
RIGHT JOINทุกแถวของตารางขวา + ข้อมูลตารางซ้ายที่ match
FULL OUTER JOINทุกแถวจากทั้งสองตาราง (NULL ในส่วนที่ไม่มีคู่)
CROSS JOINCartesian Product — ทุก combination
SELF JOINJOIN ตารางเดิมกับตัวเอง

INNER JOIN

-- ดูคำสั่งซื้อพร้อมชื่อลูกค้า SELECT o.id AS order_id, c.name AS customer, o.order_date, o.total FROM orders o INNER JOIN customers c ON o.customer_id = c.id WHERE o.status = 'paid' ORDER BY o.order_date DESC;

LEFT JOIN — ค้นหาข้อมูลที่ขาดหาย

-- ลูกค้าทุกคน รวมถึงผู้ที่ยังไม่เคยสั่งซื้อ SELECT c.name, COUNT(o.id) AS order_count FROM customers c LEFT JOIN orders o ON c.id = o.customer_id GROUP BY c.id, c.name ORDER BY order_count DESC; -- หาลูกค้าที่ไม่มีออเดอร์เลย SELECT c.name FROM customers c LEFT JOIN orders o ON c.id = o.customer_id WHERE o.id IS NULL;

Multi-table JOIN และ SELF JOIN

-- เชื่อม 3 ตาราง SELECT c.name AS customer, p.name AS product, oi.quantity, oi.unit_price FROM orders o JOIN customers c ON o.customer_id = c.id JOIN order_items oi ON o.id = oi.order_id JOIN products p ON oi.product_id = p.id; -- SELF JOIN (Employee + Manager) SELECT e.name AS employee, m.name AS manager FROM employees e LEFT JOIN employees m ON e.manager_id = m.id;

GROUP BY และ HAVING

-- ยอดขายรวมต่อ category SELECT category, COUNT(*) AS product_count, AVG(price) AS avg_price, MIN(price) AS min_price, MAX(price) AS max_price FROM products GROUP BY category ORDER BY avg_price DESC; -- HAVING: กรองกลุ่ม (ต่างจาก WHERE ที่กรองแถว) SELECT customer_id, COUNT(*) AS order_count, SUM(total) AS total_spent FROM orders WHERE status = 'paid' -- กรองแถวก่อน GROUP BY GROUP BY customer_id HAVING SUM(total) > 10000 -- กรองหลัง GROUP BY ORDER BY total_spent DESC; -- COUNT(*) vs COUNT(column) COUNT(*) -- นับทุกแถว รวม NULL COUNT(email) -- นับเฉพาะแถวที่ email ไม่ใช่ NULL COUNT(DISTINCT city) -- นับค่าไม่ซ้ำ

ROLLUP — รวมย่อย

-- สร้างรายงานพร้อม subtotal และ grand total SELECT COALESCE(category, '-- รวมทั้งหมด --') AS category, SUM(price) AS total FROM products GROUP BY category WITH ROLLUP;

บทที่ 8 — Subquery และ CTE (WITH clause)

Correlated Subquery, EXISTS, IN, CTE, Recursive CTE

Subquery ใน WHERE

-- สินค้าที่ราคาสูงกว่าค่าเฉลี่ย SELECT name, price FROM products WHERE price > ( SELECT AVG(price) FROM products ); -- ลูกค้าที่เคยสั่งซื้อสินค้า Electronics SELECT name FROM customers WHERE id IN ( SELECT DISTINCT o.customer_id FROM orders o JOIN order_items oi ON o.id = oi.order_id JOIN products p ON oi.product_id = p.id WHERE p.category = 'Electronics' );

EXISTS และ Correlated Subquery

-- EXISTS: ตรวจสอบว่ามีแถวอยู่หรือไม่ SELECT name FROM customers c WHERE EXISTS ( SELECT 1 FROM orders o WHERE o.customer_id = c.id -- อ้างอิง outer query AND o.total > 5000 ); -- Subquery ใน FROM (Derived Table) SELECT category, avg_price FROM ( SELECT category, AVG(price) AS avg_price FROM products GROUP BY category ) AS cat_avg WHERE avg_price > 1000;

CTE (Common Table Expression) — WITH clause

-- CTE ทำให้ query อ่านง่ายขึ้น WITH top_customers AS ( SELECT customer_id, SUM(total) AS total_spent FROM orders WHERE status = 'paid' GROUP BY customer_id HAVING SUM(total) > 50000 ), customer_details AS ( SELECT tc.*, c.name, c.city FROM top_customers tc JOIN customers c ON tc.customer_id = c.id ) SELECT * FROM customer_details ORDER BY total_spent DESC; -- Recursive CTE (สร้าง hierarchy) WITH RECURSIVE org_chart AS ( -- Base case: CEO (ไม่มี manager) SELECT id, name, manager_id, 0 AS level FROM employees WHERE manager_id IS NULL UNION ALL -- Recursive: หาผู้ใต้บังคับบัญชา SELECT e.id, e.name, e.manager_id, oc.level + 1 FROM employees e JOIN org_chart oc ON e.manager_id = oc.id ) SELECT CONCAT(REPEAT(' ', level), name) AS hierarchy FROM org_chart;

บทที่ 9 — Window Functions

OVER, PARTITION BY, ROW_NUMBER, RANK, LAG/LEAD, Running Totals

Window Function คืออะไร?

คำนวณค่าข้ามหลายแถว โดยไม่ยุบแถว (ต่างจาก GROUP BY ที่ยุบเป็นแถวเดียว)

GROUP BY (ยุบแถว)
SELECT dept, AVG(salary) FROM employees GROUP BY dept; -- ได้ 1 แถวต่อ dept
WINDOW (ไม่ยุบ)
SELECT name, dept, salary, AVG(salary) OVER( PARTITION BY dept ) AS dept_avg FROM employees; -- ทุกแถวยังอยู่ครบ

Ranking Functions

SELECT name, salary, dept, -- ROW_NUMBER: ลำดับไม่ซ้ำ ROW_NUMBER() OVER(ORDER BY salary DESC) AS row_num, -- RANK: ลำดับที่เท่ากันข้าม RANK() OVER(ORDER BY salary DESC) AS rnk, -- DENSE_RANK: ลำดับที่เท่ากันไม่ข้าม DENSE_RANK() OVER(ORDER BY salary DESC) AS dense_rnk, -- NTILE: แบ่งเป็น N กลุ่ม NTILE(4) OVER(ORDER BY salary) AS quartile, -- อันดับใน dept RANK() OVER(PARTITION BY dept ORDER BY salary DESC) AS dept_rank FROM employees;

LAG, LEAD — เปรียบเทียบกับแถวก่อน/หลัง

SELECT order_date, total, -- ยอดคำสั่งซื้อก่อนหน้า LAG(total) OVER(ORDER BY order_date) AS prev_total, -- ยอดคำสั่งซื้อถัดไป LEAD(total) OVER(ORDER BY order_date) AS next_total, -- % เปลี่ยนแปลง (total - LAG(total) OVER(ORDER BY order_date)) / NULLIF(LAG(total) OVER(ORDER BY order_date), 0) * 100 AS pct_change FROM orders WHERE status = 'paid' ORDER BY order_date;

Running Total และ Moving Average

-- Cumulative Sum (Running Total) SELECT order_date, total, SUM(total) OVER( ORDER BY order_date ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW ) AS running_total, -- Moving Average (7 วัน) AVG(total) OVER( ORDER BY order_date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW ) AS moving_avg_7d FROM orders ORDER BY order_date; -- TOP N per group (ใช้ CTE) WITH ranked AS ( SELECT *, ROW_NUMBER() OVER( PARTITION BY category ORDER BY price DESC ) AS rn FROM products ) SELECT * FROM ranked WHERE rn <= 3; -- Top 3 สินค้าแพงสุดต่อ category

บทที่ 10 — Index และ Query Performance

การสร้าง Index, EXPLAIN, และเทคนิค optimize

Index คืออะไร?

โครงสร้างข้อมูลพิเศษ (มักเป็น B-Tree) ที่ช่วยให้ค้นหาข้อมูลได้เร็วขึ้นมาก แทนที่จะสแกนทุกแถว (Full Table Scan)

❌ ไม่มี Index (Full Scan)
สแกนทุกแถว = O(n)
1 ล้านแถว → ช้ามาก
✅ มี Index (B-Tree)
ค้นหาแบบ Binary = O(log n)
1 ล้านแถว → ~20 steps

สร้างและลบ Index

-- Single column index CREATE INDEX idx_email ON customers (email); -- Unique index CREATE UNIQUE INDEX idx_email_unique ON customers (email); -- Composite index (หลาย column) CREATE INDEX idx_order_status_date ON orders (status, order_date); -- Partial index (PostgreSQL) CREATE INDEX idx_active_orders ON orders (customer_id) WHERE status = 'paid'; -- ลบ index DROP INDEX idx_email ON customers;

EXPLAIN — วิเคราะห์ Query

-- ดูว่า MySQL ใช้ index หรือเปล่า EXPLAIN SELECT * FROM orders WHERE customer_id = 5; -- EXPLAIN ANALYZE (PostgreSQL) EXPLAIN ANALYZE SELECT * FROM orders WHERE customer_id = 5; -- columns สำคัญใน EXPLAIN: -- type: ALL(แย่) → ref → eq_ref → const(ดีสุด) -- key: index ที่ใช้ -- rows: ประมาณแถวที่สแกน -- Extra: Using index, Using filesort ฯลฯ

บทที่ 11 — Transactions และ ACID

COMMIT, ROLLBACK, SAVEPOINT, Isolation Levels

ACID Properties

A — Atomicity
ทุกอย่างสำเร็จหรือล้มเหลวพร้อมกัน ไม่มีทำครึ่งเดียว
C — Consistency
ข้อมูลต้องถูกต้องตาม constraint ก่อนและหลัง transaction
I — Isolation
Transaction ที่ทำพร้อมกันไม่กระทบกัน
D — Durability
เมื่อ COMMIT แล้ว ข้อมูลถูกบันทึกถาวร แม้ระบบล่ม

ตัวอย่าง Transaction: โอนเงิน

-- เริ่ม transaction START TRANSACTION; -- หรือ BEGIN UPDATE accounts SET balance = balance - 1000 WHERE id = 1; -- หักจากบัญชี A -- ตรวจสอบยอดเงิน SELECT balance FROM accounts WHERE id = 1; UPDATE accounts SET balance = balance + 1000 WHERE id = 2; -- เพิ่มให้บัญชี B -- ถ้าสำเร็จ → บันทึก COMMIT; -- ถ้าผิดพลาด → ย้อนกลับ ROLLBACK; -- SAVEPOINT: บันทึกจุดกลางทาง START TRANSACTION; INSERT INTO orders ...; SAVEPOINT after_order; INSERT INTO order_items ...; -- ถ้า items ผิด → ย้อนแค่ items ROLLBACK TO SAVEPOINT after_order; COMMIT;

บทที่ 12 — Database Design & Normalization

Normal Forms, Foreign Keys, Views, Stored Procedures

Normal Forms

1NF
First Normal Form — ทุก cell มีค่าเดียว ไม่มี repeating groups เช่น ไม่เก็บ "กรุงเทพ, เชียงใหม่" ในช่องเดียว
↓
2NF
Second Normal Form — ผ่าน 1NF + ทุก non-key column ขึ้นกับ primary key ทั้งหมด (ไม่มี Partial Dependency)
↓
3NF
Third Normal Form — ผ่าน 2NF + ไม่มี Transitive Dependency (column A → column B → column C)
↓
BCNF
Boyce-Codd Normal Form — เวอร์ชัน strict กว่า 3NF สำหรับกรณีพิเศษ

VIEW — ตารางเสมือน

-- สร้าง View CREATE VIEW v_customer_summary AS SELECT c.id, c.name, COUNT(o.id) AS total_orders, COALESCE(SUM(o.total), 0) AS total_spent FROM customers c LEFT JOIN orders o ON c.id = o.customer_id GROUP BY c.id, c.name; -- ใช้ View เหมือนตาราง SELECT * FROM v_customer_summary WHERE total_spent > 10000; -- แก้ไข View CREATE OR REPLACE VIEW v_customer_summary AS ...; -- ลบ View DROP VIEW IF EXISTS v_customer_summary;
